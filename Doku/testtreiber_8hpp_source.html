56 template <typename T>
57 bool Testtreiber::test_min_heap_correctness(Heap<T>& heap_variable, int start_index, int heap_size) {
58  bool heap_correct = true;
59  int left_child_index = Heap<T>::index_of_left_child(start_index);
60  bool left_child_exists = heap_size > left_child_index;
61  int right_child_index = Heap<T>::index_of_right_child(start_index);
62  bool right_child_exists = heap_size > right_child_index;
63  int parent_index = Heap<T>::index_of_parent(start_index);
64  bool parent_exists = true;
65  if (parent_index == -1)
66  parent_exists = false;
67  if (left_child_exists)
68  heap_correct = test_min_heap_correctness(heap_variable, left_child_index, heap_size);
69  if (heap_correct && right_child_exists)
70  heap_correct = test_min_heap_correctness(heap_variable, right_child_index, heap_size);
71  try {
72  if (heap_correct && parent_exists &&
73  heap_variable.get_value_at(start_index) < heap_variable.get_value_at(parent_index)) {
74  heap_correct = false;
75  }
76  } catch (IndexOutOfBoundsException&) {
77  }
78  return heap_correct;
79 }
